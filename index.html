Initial setup Supabaseで最初に作るもの 表示する 閉じる

SupabaseのSQL Editorに下のSQLを貼り付けて実行してください。ログインした本人の行だけ読める設定です。

create table if not exists public.kusayakyu_sync_data (
  user_id uuid not null references auth.users(id) on delete cascade,
  profile_id text not null default 'default',
  data jsonb not null default '{"games":[],"plateAppearances":[]}'::jsonb,
  updated_at timestamptz not null default now(),
  primary key (user_id, profile_id)
);

alter table public.kusayakyu_sync_data enable row level security;

drop policy if exists "Own sync rows can be selected" on public.kusayakyu_sync_data;
create policy "Own sync rows can be selected"
on public.kusayakyu_sync_data for select
to authenticated
using ((select auth.uid()) = user_id);

drop policy if exists "Own sync rows can be inserted" on public.kusayakyu_sync_data;
create policy "Own sync rows can be inserted"
on public.kusayakyu_sync_data for insert
to authenticated
with check ((select auth.uid()) = user_id);

drop policy if exists "Own sync rows can be updated" on public.kusayakyu_sync_data;
create policy "Own sync rows can be updated"
on public.kusayakyu_sync_data for update
to authenticated
using ((select auth.uid()) = user_id)
with check ((select auth.uid()) = user_id);
